25 void
26 glue_conv::apply(Mat<eT>& out, const Mat<eT>& A, const Mat<eT>& B, const bool A_is_col)
27  {
28  arma_extra_debug_sigprint();
29 
30  const Mat<eT>& h = (A.n_elem <= B.n_elem) ? A : B;
31  const Mat<eT>& x = (A.n_elem <= B.n_elem) ? B : A;
32 
33  const uword h_n_elem = h.n_elem;
34  const uword h_n_elem_m1 = h_n_elem - 1;
35  const uword x_n_elem = x.n_elem;
36  const uword out_n_elem = ((h_n_elem + x_n_elem) > 0) ? (h_n_elem + x_n_elem - 1) : uword(0);
37 
38  if( (h_n_elem == 0) || (x_n_elem == 0) ) { out.zeros(); return; }
39 
40 
41  Col<eT> hh(h_n_elem, arma_nozeros_indicator()); // flipped version of h
42 
43  const eT* h_mem = h.memptr();
44  eT* hh_mem = hh.memptr();
45 
46  for(uword i=0; i < h_n_elem; ++i)
47  {
48  hh_mem[h_n_elem_m1-i] = h_mem[i];
49  }
50 
51 
52  Col<eT> xx( (x_n_elem + 2*h_n_elem_m1), arma_zeros_indicator() ); // zero padded version of x
53 
54  const eT* x_mem = x.memptr();
55  eT* xx_mem = xx.memptr();
56 
57  arrayops::copy( &(xx_mem[h_n_elem_m1]), x_mem, x_n_elem );
58 
59 
60  (A_is_col) ? out.set_size(out_n_elem, 1) : out.set_size(1, out_n_elem);
61 
62  eT* out_mem = out.memptr();
63 
64  for(uword i=0; i < out_n_elem; ++i)
65  {
66  // out_mem[i] = dot( hh, xx.subvec(i, (i + h_n_elem_m1)) );
67 
68  out_mem[i] = op_dot::direct_dot( h_n_elem, hh_mem, &(xx_mem[i]) );
69  }
70  }

168 void
169 glue_conv::apply(Mat<typename T1::elem_type>& out, const Glue<T1,T2,glue_conv>& expr)
170  {
171  arma_extra_debug_sigprint();
172 
173  typedef typename T1::elem_type eT;
174 
175  const quasi_unwrap<T1> UA(expr.A);
176  const quasi_unwrap<T2> UB(expr.B);
177 
178  const Mat<eT>& A = UA.M;
179  const Mat<eT>& B = UB.M;
180 
181  arma_debug_check
182  (
183  ( ((A.is_vec() == false) && (A.is_empty() == false)) || ((B.is_vec() == false) && (B.is_empty() == false)) ),
184  "conv(): given object must be a vector"
185  );
186 
187  const bool A_is_col = ((T1::is_col) || (A.n_cols == 1));
188 
189  const uword mode = expr.aux_uword;
190 
191  if(mode == 0) // full convolution
192  {
193  glue_conv::apply(out, A, B, A_is_col);
194  }
195  else
196  if(mode == 1) // same size as A
197  {
198  Mat<eT> tmp;
199 
200  glue_conv::apply(tmp, A, B, A_is_col);
201 
202  if( (tmp.is_empty() == false) && (A.is_empty() == false) && (B.is_empty() == false) )
203  {
204  const uword start = uword( std::floor( double(B.n_elem) / double(2) ) );
205 
206  out = (A_is_col) ? tmp(start, 0, arma::size(A)) : tmp(0, start, arma::size(A));
207  }
208  else
209  {
210  out.zeros( arma::size(A) );
211  }
212  }
213  }

224 void
225 glue_conv2::apply(Mat<eT>& out, const Mat<eT>& A, const Mat<eT>& B)
226  {
227  arma_extra_debug_sigprint();
228 
229  const Mat<eT>& G = (A.n_elem <= B.n_elem) ? A : B; // unflipped filter coefficients
230  const Mat<eT>& W = (A.n_elem <= B.n_elem) ? B : A; // original 2D image
231 
232  const uword out_n_rows = ((W.n_rows + G.n_rows) > 0) ? (W.n_rows + G.n_rows - 1) : uword(0);
233  const uword out_n_cols = ((W.n_cols + G.n_cols) > 0) ? (W.n_cols + G.n_cols - 1) : uword(0);
234 
235  if(G.is_empty() || W.is_empty()) { out.zeros(); return; }
236 
237 
238  Mat<eT> H(G.n_rows, G.n_cols, arma_nozeros_indicator()); // flipped filter coefficients
239 
240  const uword H_n_rows = H.n_rows;
241  const uword H_n_cols = H.n_cols;
242 
243  const uword H_n_rows_m1 = H_n_rows - 1;
244  const uword H_n_cols_m1 = H_n_cols - 1;
245 
246  for(uword col=0; col < H_n_cols; ++col)
247  {
248  eT* H_colptr = H.colptr(H_n_cols_m1 - col);
249  const eT* G_colptr = G.colptr(col);
250 
251  for(uword row=0; row < H_n_rows; ++row)
252  {
253  H_colptr[H_n_rows_m1 - row] = G_colptr[row];
254  }
255  }
256 
257 
258  Mat<eT> X( (W.n_rows + 2*H_n_rows_m1), (W.n_cols + 2*H_n_cols_m1), arma_zeros_indicator() );
259 
260  X( H_n_rows_m1, H_n_cols_m1, arma::size(W) ) = W; // zero padded version of 2D image
261 
262 
263  out.set_size( out_n_rows, out_n_cols );
264 
265  for(uword col=0; col < out_n_cols; ++col)
266  {
267  eT* out_colptr = out.colptr(col);
268 
269  for(uword row=0; row < out_n_rows; ++row)
270  {
271  // out.at(row, col) = accu( H % X(row, col, size(H)) );
272 
273  eT acc = eT(0);
274 
275  for(uword H_col = 0; H_col < H_n_cols; ++H_col)
276  {
277  const eT* X_colptr = X.colptr(col + H_col);
278 
279  acc += op_dot::direct_dot( H_n_rows, H.colptr(H_col), &(X_colptr[row]) );
280  }
281 
282  out_colptr[row] = acc;
283  }
284  }
285  }
