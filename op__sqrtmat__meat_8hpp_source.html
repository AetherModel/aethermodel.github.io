48 inline
49 bool
50 op_sqrtmat::apply_direct(Mat< std::complex<typename T1::elem_type> >& out, const Op<T1,op_diagmat>& expr)
51  {
52  arma_extra_debug_sigprint();
53 
54  typedef typename T1::elem_type T;
55 
56  const diagmat_proxy<T1> P(expr.m);
57 
58  arma_debug_check( (P.n_rows != P.n_cols), "sqrtmat(): given matrix must be square sized" );
59 
60  const uword N = P.n_rows;
61 
62  out.zeros(N,N);
63 
64  bool singular = false;
65 
66  for(uword i=0; i<N; ++i)
67  {
68  const T val = P[i];
69 
70  if(val >= T(0))
71  {
72  singular = (singular || (val == T(0)));
73 
74  out.at(i,i) = std::sqrt(val);
75  }
76  else
77  {
78  out.at(i,i) = std::sqrt( std::complex<T>(val) );
79  }
80  }
81 
82  return (singular) ? false : true;
83  }

88 inline
89 bool
90 op_sqrtmat::apply_direct(Mat< std::complex<typename T1::elem_type> >& out, const Base<typename T1::elem_type,T1>& expr)
91  {
92  arma_extra_debug_sigprint();
93 
94  typedef typename T1::elem_type in_T;
95  typedef typename std::complex<in_T> out_T;
96 
97  const quasi_unwrap<T1> expr_unwrap(expr.get_ref());
98  const Mat<in_T>& A = expr_unwrap.M;
99 
100  arma_debug_check( (A.is_square() == false), "sqrtmat(): given matrix must be square sized" );
101 
102  if(A.n_elem == 0)
103  {
104  out.reset();
105  return true;
106  }
107  else
108  if(A.n_elem == 1)
109  {
110  out.set_size(1,1);
111  out[0] = std::sqrt( std::complex<in_T>( A[0] ) );
112  return true;
113  }
114 
115  if(A.is_diagmat())
116  {
117  const uword N = A.n_rows;
118 
119  out.zeros(N,N); // aliasing can't happen as op_sqrtmat is defined as cx_mat = op(mat)
120 
121  for(uword i=0; i<N; ++i)
122  {
123  const in_T val = A.at(i,i);
124 
125  if(val >= in_T(0))
126  {
127  out.at(i,i) = std::sqrt(val);
128  }
129  else
130  {
131  out.at(i,i) = std::sqrt( out_T(val) );
132  }
133  }
134 
135  return true;
136  }
137 
138  #if defined(ARMA_OPTIMISE_SYMPD)
139  const bool try_sympd = sympd_helper::guess_sympd_anysize(A);
140  #else
141  const bool try_sympd = false;
142  #endif
143 
144  if(try_sympd)
145  {
146  arma_extra_debug_print("op_sqrtmat: attempting sympd optimisation");
147 
148  // if matrix A is sympd, all its eigenvalues are positive
149 
150  Col<in_T> eigval;
151  Mat<in_T> eigvec;
152 
153  const bool eig_status = eig_sym_helper(eigval, eigvec, A, 'd', "sqrtmat()");
154 
155  if(eig_status)
156  {
157  // ensure each eigenvalue is > 0
158 
159  const uword N = eigval.n_elem;
160  const in_T* eigval_mem = eigval.memptr();
161 
162  bool all_pos = true;
163 
164  for(uword i=0; i<N; ++i) { all_pos = (eigval_mem[i] <= in_T(0)) ? false : all_pos; }
165 
166  if(all_pos)
167  {
168  eigval = sqrt(eigval);
169 
170  out = conv_to< Mat<out_T> >::from( eigvec * diagmat(eigval) * eigvec.t() );
171 
172  return true;
173  }
174  }
175 
176  arma_extra_debug_print("op_sqrtmat: sympd optimisation failed");
177 
178  // fallthrough if eigen decomposition failed or an eigenvalue is zero
179  }
180 
181 
182  Mat<out_T> U;
183  Mat<out_T> S(A.n_rows, A.n_cols, arma_nozeros_indicator());
184 
185  const in_T* Amem = A.memptr();
186  out_T* Smem = S.memptr();
187 
188  const uword n_elem = A.n_elem;
189 
190  for(uword i=0; i<n_elem; ++i)
191  {
192  Smem[i] = std::complex<in_T>( Amem[i] );
193  }
194 
195  const bool schur_ok = auxlib::schur(U,S);
196 
197  if(schur_ok == false)
198  {
199  arma_extra_debug_print("sqrtmat(): schur decomposition failed");
200  out.soft_reset();
201  return false;
202  }
203 
204  const bool status = op_sqrtmat_cx::helper(S);
205 
206  const Mat<out_T> X = U*S;
207 
208  S.reset();
209 
210  out = X*U.t();
211 
212  return status;
213  }

402 template<typename T>
403 inline
404 bool
405 op_sqrtmat_cx::helper(Mat< std::complex<T> >& S)
406  {
407  typedef typename std::complex<T> eT;
408 
409  if(S.is_empty()) { return true; }
410 
411  const uword N = S.n_rows;
412 
413  const eT zero = eT(0);
414 
415  eT& S_00 = S[0];
416 
417  bool singular = (S_00 == zero);
418 
419  S_00 = std::sqrt(S_00);
420 
421  for(uword j=1; j < N; ++j)
422  {
423  eT* S_j = S.colptr(j);
424 
425  eT& S_jj = S_j[j];
426 
427  singular = (singular || (S_jj == zero));
428 
429  S_jj = std::sqrt(S_jj);
430 
431  for(uword ii=0; ii <= (j-1); ++ii)
432  {
433  const uword i = (j-1) - ii;
434 
435  const eT* S_i = S.colptr(i);
436 
437  //S_j[i] /= (S_i[i] + S_j[j]);
438  S_j[i] /= (S_i[i] + S_jj);
439 
440  for(uword k=0; k < i; ++k)
441  {
442  S_j[k] -= S_i[k] * S_j[i];
443  }
444  }
445  }
446 
447  return (singular) ? false : true;
448  }
