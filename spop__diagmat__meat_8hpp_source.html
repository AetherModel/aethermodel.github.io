48 inline
49 void
50 spop_diagmat::apply_noalias(SpMat<typename T1::elem_type>& out, const SpBase<typename T1::elem_type, T1>& expr)
51  {
52  arma_extra_debug_sigprint();
53 
54  typedef typename T1::elem_type eT;
55 
56  const SpProxy<T1> P(expr.get_ref());
57 
58  const uword P_n_rows = P.get_n_rows();
59  const uword P_n_cols = P.get_n_cols();
60  const uword P_n_nz = P.get_n_nonzero();
61 
62  const bool P_is_vec = (P_n_rows == 1) || (P_n_cols == 1);
63 
64  if(P_is_vec) // generate a diagonal matrix out of a vector
65  {
66  const uword N = (P_n_rows == 1) ? P_n_cols : P_n_rows;
67 
68  out.zeros(N, N);
69 
70  if(P_n_nz == 0) { return; }
71 
72  typename SpProxy<T1>::const_iterator_type it = P.begin();
73 
74  if(P_n_cols == 1)
75  {
76  for(uword i=0; i < P_n_nz; ++i)
77  {
78  const uword row = it.row();
79 
80  out.at(row,row) = (*it);
81 
82  ++it;
83  }
84  }
85  else
86  if(P_n_rows == 1)
87  {
88  for(uword i=0; i < P_n_nz; ++i)
89  {
90  const uword col = it.col();
91 
92  out.at(col,col) = (*it);
93 
94  ++it;
95  }
96  }
97  }
98  else // generate a diagonal matrix out of a matrix
99  {
100  out.zeros(P_n_rows, P_n_cols);
101 
102  const uword N = (std::min)(P_n_rows, P_n_cols);
103 
104  if( (is_SpMat<typename SpProxy<T1>::stored_type>::value) && (P_n_nz >= 5*N) )
105  {
106  const unwrap_spmat<typename SpProxy<T1>::stored_type> U(P.Q);
107 
108  const SpMat<eT>& X = U.M;
109 
110  for(uword i=0; i < N; ++i)
111  {
112  const eT val = X.at(i,i); // use binary search
113 
114  if(val != eT(0)) { out.at(i,i) = val; }
115  }
116  }
117  else
118  {
119  if(P_n_nz == 0) { return; }
120 
121  typename SpProxy<T1>::const_iterator_type it = P.begin();
122 
123  for(uword i=0; i < P_n_nz; ++i)
124  {
125  const uword row = it.row();
126  const uword col = it.col();
127 
128  if(row == col) { out.at(row,row) = (*it); }
129 
130  ++it;
131  }
132  }
133  }
134  }

276 spop_diagmat::apply_noalias(SpMat<typename T1::elem_type>& out, const SpGlue<T1,T2,spglue_times>& expr)
277  {
278  arma_extra_debug_sigprint();
279 
280  typedef typename T1::elem_type eT;
281 
282  const unwrap_spmat<T1> UA(expr.A);
283  const unwrap_spmat<T2> UB(expr.B);
284 
285  const SpMat<eT>& A = UA.M;
286  const SpMat<eT>& B = UB.M;
287 
288  arma_debug_assert_mul_size(A.n_rows, A.n_cols, B.n_rows, B.n_cols, "matrix multiplication");
289 
290  const uword C_n_rows = A.n_rows;
291  const uword C_n_cols = B.n_cols;
292 
293  const bool is_vec = (C_n_rows == 1) || (C_n_cols == 1);
294 
295  if(is_vec) // generate a diagonal matrix out of a vector
296  {
297  const SpMat<eT> C = A*B;
298 
299  spop_diagmat::apply_noalias(out, C);
300  }
301  else // generate a diagonal matrix out of a matrix
302  {
303  const uword N = (std::min)(C_n_rows, C_n_cols);
304 
305  if( (A.n_nonzero >= 5*N) || (B.n_nonzero >= 5*N) )
306  {
307  out.zeros(C_n_rows, C_n_cols);
308 
309  for(uword k=0; k < N; ++k)
310  {
311  typename SpMat<eT>::const_col_iterator B_it = B.begin_col_no_sync(k);
312  typename SpMat<eT>::const_col_iterator B_it_end = B.end_col_no_sync(k);
313 
314  eT acc = eT(0);
315 
316  while(B_it != B_it_end)
317  {
318  const eT B_val = (*B_it);
319  const uword i = B_it.row();
320 
321  acc += A.at(k,i) * B_val;
322 
323  ++B_it;
324  }
325 
326  out(k,k) = acc;
327  }
328  }
329  else
330  {
331  const SpMat<eT> C = A*B;
332 
333  spop_diagmat::apply_noalias(out, C);
334  }
335  }
336  }

393 
394  out.zeros(n_elem + n_pad, n_elem + n_pad);
