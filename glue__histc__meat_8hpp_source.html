23 void
24 glue_histc::apply_noalias(Mat<uword>& C, const Mat<eT>& A, const Mat<eT>& B, const uword dim)
25  {
26  arma_extra_debug_sigprint();
27 
28  arma_debug_check( ((B.is_vec() == false) && (B.is_empty() == false)), "histc(): parameter 'edges' must be a vector" );
29 
30  const uword A_n_rows = A.n_rows;
31  const uword A_n_cols = A.n_cols;
32 
33  const uword B_n_elem = B.n_elem;
34 
35  if( B_n_elem == uword(0) ) { C.reset(); return; }
36 
37  arma_debug_check
38  (
39  ((Col<eT>(const_cast<eT*>(B.memptr()), B_n_elem, false, false)).is_sorted("strictascend") == false),
40  "hist(): given 'edges' vector does not contain monotonically increasing values"
41  );
42 
43  const eT* B_mem = B.memptr();
44  const uword B_n_elem_m1 = B_n_elem - 1;
45 
46  if(dim == uword(0))
47  {
48  C.zeros(B_n_elem, A_n_cols);
49 
50  for(uword col=0; col < A_n_cols; ++col)
51  {
52  const eT* A_coldata = A.colptr(col);
53  uword* C_coldata = C.colptr(col);
54 
55  for(uword row=0; row < A_n_rows; ++row)
56  {
57  const eT x = A_coldata[row];
58 
59  for(uword i=0; i < B_n_elem_m1; ++i)
60  {
61  if( (B_mem[i] <= x) && (x < B_mem[i+1]) ) { C_coldata[i]++; break; }
62  else if( B_mem[B_n_elem_m1] == x ) { C_coldata[B_n_elem_m1]++; break; } // for compatibility with Matlab
63  }
64  }
65  }
66  }
67  else
68  if(dim == uword(1))
69  {
70  C.zeros(A_n_rows, B_n_elem);
71 
72  if(A.n_rows == 1)
73  {
74  const uword A_n_elem = A.n_elem;
75  const eT* A_mem = A.memptr();
76  uword* C_mem = C.memptr();
77 
78  for(uword j=0; j < A_n_elem; ++j)
79  {
80  const eT x = A_mem[j];
81 
82  for(uword i=0; i < B_n_elem_m1; ++i)
83  {
84  if( (B_mem[i] <= x) && (x < B_mem[i+1]) ) { C_mem[i]++; break; }
85  else if( B_mem[B_n_elem_m1] == x ) { C_mem[B_n_elem_m1]++; break; } // for compatibility with Matlab
86  }
87  }
88  }
89  else
90  {
91  for(uword row=0; row < A_n_rows; ++row)
92  for(uword col=0; col < A_n_cols; ++col)
93  {
94  const eT x = A.at(row,col);
95 
96  for(uword i=0; i < B_n_elem_m1; ++i)
97  {
98  if( (B_mem[i] <= x) && (x < B_mem[i+1]) ) { C.at(row,i)++; break; }
99  else if( B_mem[B_n_elem_m1] == x ) { C.at(row,B_n_elem_m1)++; break; } // for compatibility with Matlab
100  }
101  }
102  }
103  }
104  }
