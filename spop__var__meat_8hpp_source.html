57 inline
58 void
59 spop_var::apply_noalias
60  (
61  SpMat<typename T1::pod_type>& out,
62  const SpProxy<T1>& p,
63  const uword norm_type,
64  const uword dim
65  )
66  {
67  arma_extra_debug_sigprint();
68 
69  typedef typename T1::elem_type in_eT;
70  //typedef typename T1::pod_type out_eT;
71 
72  const uword p_n_rows = p.get_n_rows();
73  const uword p_n_cols = p.get_n_cols();
74 
75  // TODO: this is slow; rewrite based on the approach used by sparse mean()
76 
77  if(dim == 0) // find variance in each column
78  {
79  arma_extra_debug_print("spop_var::apply_noalias(): dim = 0");
80 
81  out.set_size((p_n_rows > 0) ? 1 : 0, p_n_cols);
82 
83  if( (p_n_rows == 0) || (p.get_n_nonzero() == 0) ) { return; }
84 
85  for(uword col = 0; col < p_n_cols; ++col)
86  {
87  if(SpProxy<T1>::use_iterator)
88  {
89  // We must use an iterator; we can't access memory directly.
90  typename SpProxy<T1>::const_iterator_type it = p.begin_col(col);
91  typename SpProxy<T1>::const_iterator_type end = p.begin_col(col + 1);
92 
93  const uword n_zero = p_n_rows - (end.pos() - it.pos());
94 
95  // in_eT is used just to get the specialization right (complex / noncomplex)
96  out.at(0, col) = spop_var::iterator_var(it, end, n_zero, norm_type, in_eT(0));
97  }
98  else
99  {
100  // We can use direct memory access to calculate the variance.
101  out.at(0, col) = spop_var::direct_var
102  (
103  &p.get_values()[p.get_col_ptrs()[col]],
104  p.get_col_ptrs()[col + 1] - p.get_col_ptrs()[col],
105  p_n_rows,
106  norm_type
107  );
108  }
109  }
110  }
111  else
112  if(dim == 1) // find variance in each row
113  {
114  arma_extra_debug_print("spop_var::apply_noalias(): dim = 1");
115 
116  out.set_size(p_n_rows, (p_n_cols > 0) ? 1 : 0);
117 
118  if( (p_n_cols == 0) || (p.get_n_nonzero() == 0) ) { return; }
119 
120  for(uword row = 0; row < p_n_rows; ++row)
121  {
122  // We have to use an iterator here regardless of whether or not we can
123  // directly access memory.
124  typename SpProxy<T1>::const_row_iterator_type it = p.begin_row(row);
125  typename SpProxy<T1>::const_row_iterator_type end = p.end_row(row);
126 
127  const uword n_zero = p_n_cols - (end.pos() - it.pos());
128 
129  out.at(row, 0) = spop_var::iterator_var(it, end, n_zero, norm_type, in_eT(0));
130  }
131  }
132  }

158 template<typename eT>
159 inline
160 eT
161 spop_var::direct_var
162  (
163  const eT* const X,
164  const uword length,
165  const uword N,
166  const uword norm_type
167  )
168  {
169  arma_extra_debug_sigprint();
170 
171  if(length >= 2 && N >= 2)
172  {
173  const eT acc1 = spop_mean::direct_mean(X, length, N);
174 
175  eT acc2 = eT(0);
176  eT acc3 = eT(0);
177 
178  uword i, j;
179 
180  for(i = 0, j = 1; j < length; i += 2, j += 2)
181  {
182  const eT Xi = X[i];
183  const eT Xj = X[j];
184 
185  const eT tmpi = acc1 - Xi;
186  const eT tmpj = acc1 - Xj;
187 
188  acc2 += tmpi * tmpi + tmpj * tmpj;
189  acc3 += tmpi + tmpj;
190  }
191 
192  if(i < length)
193  {
194  const eT Xi = X[i];
195 
196  const eT tmpi = acc1 - Xi;
197 
198  acc2 += tmpi * tmpi;
199  acc3 += tmpi;
200  }
201 
202  // Now add in all zero elements.
203  acc2 += (N - length) * (acc1 * acc1);
204  acc3 += (N - length) * acc1;
205 
206  const eT norm_val = (norm_type == 0) ? eT(N - 1) : eT(N);
207  const eT var_val = (acc2 - (acc3 * acc3) / eT(N)) / norm_val;
208 
209  return var_val;
210  }
211  else if(length == 1 && N > 1) // if N == 1, then variance is zero.
212  {
213  const eT mean = X[0] / eT(N);
214  const eT val = mean - X[0];
215 
216  const eT acc2 = (val * val) + (N - length) * (mean * mean);
217  const eT acc3 = val + (N - length) * mean;
218 
219  const eT norm_val = (norm_type == 0) ? eT(N - 1) : eT(N);
220  const eT var_val = (acc2 - (acc3 * acc3) / eT(N)) / norm_val;
221 
222  return var_val;
223  }
224  else
225  {
226  return eT(0);
227  }
228  }

232 template<typename T>
233 inline
234 T
235 spop_var::direct_var
236  (
237  const std::complex<T>* const X,
238  const uword length,
239  const uword N,
240  const uword norm_type
241  )
242  {
243  arma_extra_debug_sigprint();
244 
245  typedef typename std::complex<T> eT;
246 
247  if(length >= 2 && N >= 2)
248  {
249  const eT acc1 = spop_mean::direct_mean(X, length, N);
250 
251  T acc2 = T(0);
252  eT acc3 = eT(0);
253 
254  for(uword i = 0; i < length; ++i)
255  {
256  const eT tmp = acc1 - X[i];
257 
258  acc2 += std::norm(tmp);
259  acc3 += tmp;
260  }
261 
262  // Add zero elements to sums
263  acc2 += std::norm(acc1) * T(N - length);
264  acc3 += acc1 * T(N - length);
265 
266  const T norm_val = (norm_type == 0) ? T(N - 1) : T(N);
267  const T var_val = (acc2 - std::norm(acc3) / T(N)) / norm_val;
268 
269  return var_val;
270  }
271  else if(length == 1 && N > 1) // if N == 1, then variance is zero.
272  {
273  const eT mean = X[0] / T(N);
274  const eT val = mean - X[0];
275 
276  const T acc2 = std::norm(val) + (N - length) * std::norm(mean);
277  const eT acc3 = val + T(N - length) * mean;
278 
279  const T norm_val = (norm_type == 0) ? T(N - 1) : T(N);
280  const T var_val = (acc2 - std::norm(acc3) / T(N)) / norm_val;
281 
282  return var_val;
283  }
284  else
285  {
286  return T(0); // All elements are zero
287  }
288  }

294 eT
295 spop_var::iterator_var
296  (
297  T1& it,
298  const T1& end,
299  const uword n_zero,
300  const uword norm_type,
301  const eT junk1,
302  const typename arma_not_cx<eT>::result* junk2
303  )
304  {
305  arma_extra_debug_sigprint();

307  arma_ignore(junk2);

309  T1 new_it(it); // for mean
310  // T1 backup_it(it); // in case we have to call robust iterator_var
311  eT mean = spop_mean::iterator_mean(new_it, end, n_zero, eT(0));
312 
313  eT acc2 = eT(0);
314  eT acc3 = eT(0);
315 
316  const uword it_begin_pos = it.pos();
317 
318  while(it != end)
319  {
320  const eT tmp = mean - (*it);
321 
322  acc2 += (tmp * tmp);
323  acc3 += (tmp);
324 
325  ++it;
326  }
327 
328  const uword n_nonzero = (it.pos() - it_begin_pos);
329  if(n_nonzero == 0)
330  {
331  return eT(0);
332  }
333 
334  if(n_nonzero + n_zero == 1)
335  {
336  return eT(0); // only one element
337  }
338 
339  // Add in entries for zeros.
340  acc2 += eT(n_zero) * (mean * mean);
341  acc3 += eT(n_zero) * mean;
342 
343  const eT norm_val = (norm_type == 0) ? eT(n_zero + n_nonzero - 1) : eT(n_zero + n_nonzero);
344  const eT var_val = (acc2 - (acc3 * acc3) / eT(n_nonzero + n_zero)) / norm_val;
345
