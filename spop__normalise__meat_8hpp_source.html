67 void
68 spop_normalise::apply_direct(SpMat<eT>& out, const SpMat<eT>& X, const uword p)
69  {
70  arma_extra_debug_sigprint();
71 
72  typedef typename get_pod_type<eT>::result T;
73 
74  SpMat<eT> tmp(arma_reserve_indicator(), X.n_rows, X.n_cols, X.n_nonzero);
75 
76  bool has_zero = false;
77 
78  podarray<T> norm_vals(X.n_cols);
79 
80  T* norm_vals_mem = norm_vals.memptr();
81 
82  for(uword col=0; col < X.n_cols; ++col)
83  {
84  const uword col_offset = X.col_ptrs[col ];
85  const uword next_col_offset = X.col_ptrs[col + 1];
86 
87  const eT* start_ptr = &X.values[ col_offset];
88  const eT* end_ptr = &X.values[next_col_offset];
89 
90  const uword n_elem = end_ptr - start_ptr;
91 
92  const Col<eT> fake_vec(const_cast<eT*>(start_ptr), n_elem, false, false);
93 
94  const T norm_val = norm(fake_vec, p);
95 
96  norm_vals_mem[col] = (norm_val != T(0)) ? norm_val : T(1);
97  }
98 
99  const uword N = X.n_nonzero;
100 
101  typename SpMat<eT>::const_iterator it = X.begin();
102 
103  for(uword i=0; i < N; ++i)
104  {
105  const uword row = it.row();
106  const uword col = it.col();
107 
108  const eT val = (*it) / norm_vals_mem[col];
109 
110  if(val == eT(0)) { has_zero = true; }
111 
112  access::rw(tmp.values[i]) = val;
113  access::rw(tmp.row_indices[i]) = row;
114  access::rw(tmp.col_ptrs[col + 1])++;
115 
116  ++it;
117  }
118 
119  for(uword c=0; c < tmp.n_cols; ++c)
120  {
121  access::rw(tmp.col_ptrs[c + 1]) += tmp.col_ptrs[c];
122  }
123 
124  if(has_zero) { tmp.remove_zeros(); }
125 
126  out.steal_mem(tmp);
127  }
