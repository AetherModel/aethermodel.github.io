53 template<typename eT, typename T1, typename T2>
54 arma_hot
55 inline
56 void
57 spglue_schur::apply_noalias(SpMat<eT>& out, const SpProxy<T1>& pa, const SpProxy<T2>& pb)
58  {
59  arma_extra_debug_sigprint();
60 
61  arma_debug_assert_same_size(pa.get_n_rows(), pa.get_n_cols(), pb.get_n_rows(), pb.get_n_cols(), "element-wise multiplication");
62 
63  if( (pa.get_n_nonzero() == 0) || (pb.get_n_nonzero() == 0) )
64  {
65  out.zeros(pa.get_n_rows(), pa.get_n_cols());
66  return;
67  }
68 
69  const uword max_n_nonzero = (std::min)(pa.get_n_nonzero(), pb.get_n_nonzero());
70 
71  // Resize memory to upper bound
72  out.reserve(pa.get_n_rows(), pa.get_n_cols(), max_n_nonzero);
73 
74  // Now iterate across both matrices.
75  typename SpProxy<T1>::const_iterator_type x_it = pa.begin();
76  typename SpProxy<T1>::const_iterator_type x_end = pa.end();
77 
78  typename SpProxy<T2>::const_iterator_type y_it = pb.begin();
79  typename SpProxy<T2>::const_iterator_type y_end = pb.end();
80 
81  uword count = 0;
82 
83  while( (x_it != x_end) || (y_it != y_end) )
84  {
85  const uword x_it_row = x_it.row();
86  const uword x_it_col = x_it.col();
87 
88  const uword y_it_row = y_it.row();
89  const uword y_it_col = y_it.col();
90 
91  if(x_it == y_it)
92  {
93  const eT out_val = (*x_it) * (*y_it);
94 
95  if(out_val != eT(0))
96  {
97  access::rw(out.values[count]) = out_val;
98 
99  access::rw(out.row_indices[count]) = x_it_row;
100  access::rw(out.col_ptrs[x_it_col + 1])++;
101  ++count;
102  }
103 
104  ++x_it;
105  ++y_it;
106  }
107  else
108  {
109  if((x_it_col < y_it_col) || ((x_it_col == y_it_col) && (x_it_row < y_it_row))) // if y is closer to the end
110  {
111  ++x_it;
112  }
113  else
114  {
115  ++y_it;
116  }
117  }
118 
119  arma_check( (count > max_n_nonzero), "internal error: spglue_schur::apply_noalias(): count > max_n_nonzero" );
120  }
121 
122  const uword out_n_cols = out.n_cols;
123 
124  uword* col_ptrs = access::rwp(out.col_ptrs);
125 
126  // Fix column pointers to be cumulative.
127  for(uword c = 1; c <= out_n_cols; ++c)
128  {
129  col_ptrs[c] += col_ptrs[c - 1];
130  }
131 
132  if(count < max_n_nonzero)
133  {
134  if(count <= (max_n_nonzero/2))
135  {
136  out.mem_resize(count);
137  }
138  else
139  {
140  // quick resize without reallocating memory and copying data
141  access::rw( out.n_nonzero) = count;
142  access::rw( out.values[count]) = eT(0);
143  access::rw(out.row_indices[count]) = uword(0);
144  }
145  }
146  }
