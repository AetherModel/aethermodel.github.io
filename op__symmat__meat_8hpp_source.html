23 inline
24 void
25 op_symmatu::apply(Mat<typename T1::elem_type>& out, const Op<T1,op_symmatu>& in)
26  {
27  arma_extra_debug_sigprint();
28 
29  typedef typename T1::elem_type eT;
30 
31  const unwrap<T1> tmp(in.m);
32  const Mat<eT>& A = tmp.M;
33 
34  arma_debug_check( (A.is_square() == false), "symmatu(): given matrix must be square sized" );
35 
36  const uword N = A.n_rows;
37 
38  if(&out != &A)
39  {
40  out.copy_size(A);
41 
42  // upper triangular: copy the diagonal and the elements above the diagonal
43 
44  for(uword i=0; i<N; ++i)
45  {
46  const eT* A_data = A.colptr(i);
47  eT* out_data = out.colptr(i);
48 
49  arrayops::copy( out_data, A_data, i+1 );
50  }
51  }
52 
53 
54  // reflect elements across the diagonal from upper triangle to lower triangle
55 
56  for(uword col=1; col < N; ++col)
57  {
58  const eT* coldata = out.colptr(col);
59  eT* row_ptr = &(out.at(col,0));
60 
61  for(uword row=0; row < col; ++row)
62  {
63  // out.at(col,row) = coldata[row];
64 
65  (*row_ptr) = coldata[row];
66  row_ptr += N;
67  }
68  }
69  }
