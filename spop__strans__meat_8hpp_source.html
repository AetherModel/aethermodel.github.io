22 template<typename eT>
23 inline
24 void
25 spop_strans::apply_noalias(SpMat<eT>& B, const SpMat<eT>& A)
26  {
27  arma_extra_debug_sigprint();
28 
29  B.reserve(A.n_cols, A.n_rows, A.n_nonzero); // deliberately swapped
30 
31  if(A.n_nonzero == 0) { return; }
32 
33  // This follows the TRANSP algorithm described in
34  // 'Sparse Matrix Multiplication Package (SMMP)'
35  // (R.E. Bank and C.C. Douglas, 2001)
36 
37  const uword m = A.n_rows;
38  const uword n = A.n_cols;
39 
40  const eT* a = A.values;
41  eT* b = access::rwp(B.values);
42 
43  const uword* ia = A.col_ptrs;
44  const uword* ja = A.row_indices;
45 
46  uword* ib = access::rwp(B.col_ptrs);
47  uword* jb = access::rwp(B.row_indices);
48 
49  // // ib is already zeroed, as B is freshly constructed
50  //
51  // for(uword i=0; i < (m+1); ++i)
52  // {
53  // ib[i] = 0;
54  // }
55 
56  for(uword i=0; i < n; ++i)
57  {
58  for(uword j = ia[i]; j < ia[i+1]; ++j)
59  {
60  ib[ ja[j] + 1 ]++;
61  }
62  }
63 
64  for(uword i=0; i < m; ++i)
65  {
66  ib[i+1] += ib[i];
67  }
68 
69  for(uword i=0; i < n; ++i)
70  {
71  for(uword j = ia[i]; j < ia[i+1]; ++j)
72  {
73  const uword jj = ja[j];
74 
75  const uword ib_jj = ib[jj];
76 
77  jb[ib_jj] = i;
78 
79  b[ib_jj] = a[j];
80 
81  ib[jj]++;
82  }
83  }
84 
85  for(uword i = m-1; i >= 1; --i)
86  {
87  ib[i] = ib[i-1];
88  }
89 
90  ib[0] = 0;
91  }
