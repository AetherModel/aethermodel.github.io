31 
32  const uword X_n_rows = obj::is_row ? uword(1) : X.n_rows;
33  const uword X_n_cols = obj::is_col ? uword(1) : X.n_cols;
34 
35  out.set_size(X_n_rows * copies_per_row, X_n_cols * copies_per_col);
36 
37  if(out.n_elem == 0) { return; }
38 
39  for(uword col=0; col < X_n_cols; ++col)
40  {
41  const uword out_col_offset = col * copies_per_col;
42 
43  eT* out_colptr_first = out.colptr(out_col_offset);
44 
45  for(uword row=0; row < X_n_rows; ++row)
46  {
47  const uword out_row_offset = row * copies_per_row;
48 
49  const eT copy_value = X.at(row, col);
50 
51  for(uword row_copy=0; row_copy < copies_per_row; ++row_copy)
52  {
53  out_colptr_first[out_row_offset + row_copy] = copy_value;
54  }
55 
56  if(copies_per_col != 1)
57  {
58  for(uword col_copy=1; col_copy < copies_per_col; ++col_copy)
59  {
60  eT* out_colptr = out.colptr(out_col_offset + col_copy);
61 
62  arrayops::copy(&out_colptr[out_row_offset], &out_colptr_first[out_row_offset], copies_per_row);
63  }
