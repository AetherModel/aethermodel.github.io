27  {
28  arma_extra_debug_sigprint();
29  arma_ignore(junk);
30 
31  typedef typename obj_type::elem_type eT;
32 
33  if(is_Col<obj_type>::value)
34  {
35  arma_debug_check( (n_cols != 1), "randi(): incompatible size" );
36  }
37  else
38  if(is_Row<obj_type>::value)
39  {
40  arma_debug_check( (n_rows != 1), "randi(): incompatible size" );
41  }
42 
43  obj_type out(n_rows, n_cols, arma_nozeros_indicator());
44 
45  int a;
46  int b;
47 
48  if(param.state == 0)
49  {
50  a = 0;
51  b = arma_rng::randi<eT>::max_val();
52  }
53  else
54  if(param.state == 1)
55  {
56  a = param.a_int;
57  b = param.b_int;
58  }
59  else
60  {
61  a = int(param.a_double);
62  b = int(param.b_double);
63  }
64 
65  arma_debug_check( (a > b), "randi(): incorrect distribution parameters: a must be less than b" );
66 
67  arma_rng::randi<eT>::fill(out.memptr(), out.n_elem, a, b);
68
