309 inline
310 void
311 spop_repelem::apply(SpMat<typename T1::elem_type>& out, const SpOp<T1, spop_repelem>& in)
312  {
313  arma_extra_debug_sigprint();
314 
315  typedef typename T1::elem_type eT;
316 
317  const unwrap_spmat<T1> U(in.m);
318  const SpMat<eT>& X = U.M;
319 
320  const uword copies_per_row = in.aux_uword_a;
321  const uword copies_per_col = in.aux_uword_b;
322 
323  const uword out_n_rows = X.n_rows * copies_per_row;
324  const uword out_n_cols = X.n_cols * copies_per_col;
325  const uword out_nnz = X.n_nonzero * copies_per_row * copies_per_col;
326 
327  if( (out_n_rows > 0) && (out_n_cols > 0) && (out_nnz > 0) )
328  {
329  umat locs(2, out_nnz, arma_nozeros_indicator());
330  Col<eT> vals( out_nnz, arma_nozeros_indicator());
331 
332  uword* locs_mem = locs.memptr();
333  eT* vals_mem = vals.memptr();
334 
335  typename SpMat<eT>::const_iterator X_it = X.begin();
336  typename SpMat<eT>::const_iterator X_end = X.end();
337 
338  for(; X_it != X_end; ++X_it)
339  {
340  const uword col_base = copies_per_col * X_it.col();
341  const uword row_base = copies_per_row * X_it.row();
342 
343  const eT X_val = (*X_it);
344 
345  for(uword cols = 0; cols < copies_per_col; cols++)
346  for(uword rows = 0; rows < copies_per_row; rows++)
347  {
348  (*locs_mem) = row_base + rows; ++locs_mem;
349  (*locs_mem) = col_base + cols; ++locs_mem;
350 
351  (*vals_mem) = X_val; ++vals_mem;
352  }
353  }
354 
355  out = SpMat<eT>(locs, vals, out_n_rows, out_n_cols);
356  }
357  else
358  {
359  out.zeros(out_n_rows, out_n_cols);
360  }
361  }

508 template<typename T1>
509 inline
510 void
511 spop_diagvec::apply(SpMat<typename T1::elem_type>& out, const SpOp<T1,spop_diagvec>& in)
512  {
513  arma_extra_debug_sigprint();
514 
515  typedef typename T1::elem_type eT;
516 
517  const unwrap_spmat<T1> U(in.m);
518 
519  const SpMat<eT>& X = U.M;
520 
521  const uword a = in.aux_uword_a;
522  const uword b = in.aux_uword_b;
523 
524  const uword row_offset = (b > 0) ? a : 0;
525  const uword col_offset = (b == 0) ? a : 0;
526 
527  arma_debug_check_bounds
528  (
529  ((row_offset > 0) && (row_offset >= X.n_rows)) || ((col_offset > 0) && (col_offset >= X.n_cols)),
530  "diagvec(): requested diagonal out of bounds"
531  );
532 
533  const uword len = (std::min)(X.n_rows - row_offset, X.n_cols - col_offset);
534 
535  Col<eT> cache(len, arma_nozeros_indicator());
536  eT* cache_mem = cache.memptr();
537 
538  uword n_nonzero = 0;
539 
540  for(uword i=0; i < len; ++i)
541  {
542  const eT val = X.at(i + row_offset, i + col_offset);
543 
544  cache_mem[i] = val;
545 
546  n_nonzero += (val != eT(0)) ? uword(1) : uword(0);
547  }
548 
549  out.reserve(len, 1, n_nonzero);
550 
551  uword count = 0;
552  for(uword i=0; i < len; ++i)
553  {
554  const eT val = cache_mem[i];
555 
556  if(val != eT(0))
557  {
558  access::rw(out.row_indices[count]) = i;
559  access::rw(out.values[count]) = val;
560  ++count;
561  }
562  }
563 
564  access::rw(out.col_ptrs[0]) = 0;
565  access::rw(out.col_ptrs[1]) = n_nonzero;
566  }
