117 template<typename eT>
118 inline
119 bool
120 glue_mvnrnd::apply_noalias(Mat<eT>& out, const Mat<eT>& M, const Mat<eT>& C, const uword N)
121  {
122  arma_extra_debug_sigprint();
123 
124  Mat<eT> D;
125 
126  const bool chol_status = op_chol::apply_direct(D, C, 1); // '1' means "lower triangular"
127 
128  if(chol_status == false)
129  {
130  // C is not symmetric positive definite, so find approximate square root of C
131 
132  Col<eT> eigval; // NOTE: eT is constrained to be real (ie. float or double) in fn_mvnrnd.hpp
133  Mat<eT> eigvec;
134 
135  const bool eig_status = eig_sym_helper(eigval, eigvec, C, 'd', "mvnrnd()");
136 
137  if(eig_status == false) { return false; }
138 
139  eT* eigval_mem = eigval.memptr();
140  const uword eigval_n_elem = eigval.n_elem;
141 
142  // since we're doing an approximation, tolerate tiny negative eigenvalues
143 
144  const eT tol = eT(-100) * Datum<eT>::eps * norm(C, "fro");
145 
146  if(arma_isfinite(tol) == false) { return false; }
147 
148  for(uword i=0; i<eigval_n_elem; ++i)
149  {
150  const eT val = eigval_mem[i];
151 
152  if( (val < tol) || (arma_isfinite(val) == false) ) { return false; }
153  }
154 
155  for(uword i=0; i<eigval_n_elem; ++i) { if(eigval_mem[i] < eT(0)) { eigval_mem[i] = eT(0); } }
156 
157  Mat<eT> DD = eigvec * diagmat(sqrt(eigval));
158 
159  D.steal_mem(DD);
160  }
161 
162  out = D * randn< Mat<eT> >(M.n_rows, N);
163 
164  if(N == 1)
165  {
166  out += M;
167  }
168  else
169  if(N > 1)
170  {
171  out.each_col() += M;
172  }
173 
174  return true;
175  }
