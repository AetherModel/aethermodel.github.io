49 template<typename eT>
50 inline
51 void
52 spop_repmat::apply_noalias(SpMat<eT>& out, const uword A_n_rows, const uword A_n_cols, const SpMat<eT>& B)
53  {
54  arma_extra_debug_sigprint();
55 
56  const uword B_n_rows = B.n_rows;
57  const uword B_n_cols = B.n_cols;
58 
59  const uword out_n_nonzero = A_n_rows * A_n_cols * B.n_nonzero;
60 
61  out.reserve(A_n_rows * B_n_rows, A_n_cols * B_n_cols, out_n_nonzero);
62 
63  if(out_n_nonzero == 0) { return; }
64 
65  access::rw(out.col_ptrs[0]) = 0;
66 
67  uword count = 0;
68 
69  for(uword A_col=0; A_col < A_n_cols; ++A_col)
70  for(uword B_col=0; B_col < B_n_cols; ++B_col)
71  {
72  for(uword A_row=0; A_row < A_n_rows; ++A_row)
73  {
74  const uword out_row = A_row * B_n_rows;
75 
76  for(uword B_i = B.col_ptrs[B_col]; B_i < B.col_ptrs[B_col+1]; ++B_i)
77  {
78  access::rw(out.values[count]) = B.values[B_i];
79  access::rw(out.row_indices[count]) = out_row + B.row_indices[B_i];
80 
81  count++;
82  }
83  }
84 
85  access::rw(out.col_ptrs[A_col * B_n_cols + B_col + 1]) = count;
86  }
87  }
