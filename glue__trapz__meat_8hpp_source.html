55 glue_trapz::apply_noalias(Mat<eT>& out, const Mat<eT>& X, const Mat<eT>& Y, const uword dim)
56  {
57  arma_extra_debug_sigprint();
58 
59  arma_debug_check( (dim > 1), "trapz(): argument 'dim' must be 0 or 1" );
60 
61  arma_debug_check( ((X.is_vec() == false) && (X.is_empty() == false)), "trapz(): argument 'X' must be a vector" );
62 
63  const uword N = X.n_elem;
64 
65  if(dim == 0)
66  {
67  arma_debug_check( (N != Y.n_rows), "trapz(): length of X must equal the number of rows in Y when dim=0" );
68  }
69  else
70  if(dim == 1)
71  {
72  arma_debug_check( (N != Y.n_cols), "trapz(): length of X must equal the number of columns in Y when dim=1" );
73  }
74 
75  if(N <= 1)
76  {
77  if(dim == 0) { out.zeros(1, Y.n_cols); }
78  else if(dim == 1) { out.zeros(Y.n_rows, 1); }
79 
80  return;
81  }
82 
83  const Col<eT> vec_X( const_cast<eT*>(X.memptr()), X.n_elem, false, true );
84 
85  const Col<eT> diff_X = diff(vec_X);
86 
87  if(dim == 0)
88  {
89  const Row<eT> diff_X_t( const_cast<eT*>(diff_X.memptr()), diff_X.n_elem, false, true );
90 
91  out = diff_X_t * (0.5 * (Y.rows(0, N-2) + Y.rows(1, N-1)));
92  }
93  else
94  if(dim == 1)
95  {
96  out = (0.5 * (Y.cols(0, N-2) + Y.cols(1, N-1))) * diff_X;
97  }
