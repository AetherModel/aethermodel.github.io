49  {

51 
52  typedef typename T1::elem_type eT;
53  typedef typename T1::pod_type T;
54 
55  arma_debug_check((tol < T(0)), "orth(): tolerance must be >= 0");
56 
57  Mat<eT> A(expr.get_ref());
58 
59  Mat<eT> U;
60  Col< T> s;
61  Mat<eT> V;
62 
63  const bool status = auxlib::svd_dc(U, s, V, A);
64 
65  V.reset();
66 
67  if(status == false) { return false; }
68 
69  if(s.is_empty()) { out.reset(); return true; }
70 
71  const uword s_n_elem = s.n_elem;
72  const T* s_mem = s.memptr();
73 
74  // set tolerance to default if it hasn't been specified
75  if(tol == T(0)) { tol = (std::max)(A.n_rows, A.n_cols) * s_mem[0] * std::numeric_limits<T>::epsilon(); }
76 
77  uword count = 0;
78 
79  for(uword i=0; i < s_n_elem; ++i) { count += (s_mem[i] > tol) ? uword(1) : uword(0); }
80 
81  if(count > 0)
82  {
83  out = U.head_cols(count); // out *= eT(-1);
84  }
85  else
86  {

88  }

152 
153  uword count = 0;
154

156

158  {
159  out = V.tail_cols(A.n_cols - count);

171  out.set_size(A.n_cols, 0);
