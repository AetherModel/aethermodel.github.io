36 inline
37 typename get_pod_type<eT>::result
38 spop_norm::mat_norm_2(const SpMat<eT>& X, const typename arma_real_only<eT>::result* junk)
39  {
40  arma_extra_debug_sigprint();
41  arma_ignore(junk);
42 
43  // norm = sqrt( largest eigenvalue of (A^H)*A ), where ^H is the conjugate transpose
44  // http://math.stackexchange.com/questions/4368/computing-the-largest-eigenvalue-of-a-very-large-sparse-matrix
45 
46  typedef typename get_pod_type<eT>::result T;
47 
48  const SpMat<eT>& A = X;
49  const SpMat<eT> B = trans(A);
50 
51  const SpMat<eT> C = (A.n_rows <= A.n_cols) ? (A*B) : (B*A);
52 
53  Col<T> eigval;
54  eigs_sym(eigval, C, 1);
55 
56  return (eigval.n_elem > 0) ? T(std::sqrt(eigval[0])) : T(0);
57  }
